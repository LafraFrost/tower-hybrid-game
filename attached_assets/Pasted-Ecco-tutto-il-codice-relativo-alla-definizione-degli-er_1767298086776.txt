2. Struttura dei Dati (interfaces.ts)
Qui è definita l'interfaccia IUnit che rappresenta l'eroe quando è in gioco (es. durante il combattimento).
code
TypeScript
export interface IUnit {
    id: string; // "Ombra", "Baluardo", "Fante_1"
    type: 'hero' | 'enemy';
    heroClass?: string; // "Tank", "DPS", "Control", "Support", "Specialist"
    currentHP: number;
    maxHP: number;
    currentPA?: number; // Solo per eroi
    currentR2?: number; // Solo per eroi
    gridCoords: string; // "x,y" (es. "3,2")
    activeBuffs: string[]; // Es. ["Difesa+3", "Veleno"]
    heroName?: string;
    isReady?: boolean;
}
3. Logica di Caricamento nel Player (PlayerHUD.tsx)
Questo snippet mostra come vengono caricati i dati dell'eroe quando un giocatore effettua il "Login" selezionando il personaggio.
code
TypeScript
// Stato locale del giocatore nel HUD
interface ILocalPlayerState {
    heroName: string;
    currentHP: number;
    maxHP: number;
    currentPA: number;
    maxPA: number;
    currentR2: number;
    maxR2: number;
    hand: Array<{ id: string, name: string, paCost: number, type: string }>;
}

// ...

const handleLogin = (hero: string) => {
    // Recupera le stats da GameData
    const stats = GAME_DATA.HERO_STATS[hero as keyof typeof GAME_DATA.HERO_STATS];
    
    setPlayerState({
        heroName: hero,
        currentHP: stats.HP_BASE,
        maxHP: stats.HP_BASE,
        currentPA: 3, // Valore fisso iniziale per prototipo
        maxPA: 3,
        currentR2: 0, // Inizia con 0 risorsa speciale
        maxR2: stats.R2_MAX,
        hand: []
    });
    setSelectedHero(hero);
    sendAction('JOIN_LOBBY', { heroName: hero });
};
